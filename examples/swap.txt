def swap(new_owner, prev_owner, c_token, c_range, block_number, tx) := 
  with SU() as c_su {
    assert(c_su.address, self.address)
    and assert(c_su.0, c_token)
    and assert(c_su.1, c_range)
    and assert(c_su.2, block_number)
    and assert(c_su.3.0, prev_owner)
    and assert(c_su.3.1, new_owner)
    and IncludedAt(c_su, c_su.0, c_su.1, c_su.2) and SignedBy(tx, new_owner)
  }
